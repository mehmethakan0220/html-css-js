Nesnelerde prototip tanımlama ve kullanma 

   function sayilar(birincisayi,ikincisayi,ucuncusayi,dorduncusayi){
     this.sayi1=birincisayi;
     this.sayi2=ikincisayi;
     this.sayi3=ucuncusayi;
     this.sayi4=dorduncusayi;
   }

   sayilar.prototype.topla = function () {
     return (this.sayi1 + this.sayi2 + this.sayi3 + this.sayi4 );
   }

   sayilar.prototype.cikar = function () {
     return (this.sayi1-this.sayi2-this.sayi3-this.sayi4);
   }

   sayilar.prototype.carp = function () {
     return (this.sayi1*this.sayi2*this.sayi3*this.sayi4);
   }

   var islem = new sayilar(1,2,3,4);

   toplam = islem.topla();
   fark = islem.cikar();
   carpim = islem.carp();

   document.write(toplam);
   document.write(fark);
   document.write(carpim);
